Try Bar Chart

bps_mz2$shortName <- str_wrap(bps_mz2$bps_name, width = 30)


bpsChart <- 
  ggplot(data = bps_mz2, aes(x = reorder(shortName,acres), y = acres)) +
  geom_bar(stat = "identity") +
  labs(
    title = "",
    subtitle = "",
    caption = "",
    x = "",
    y = "Acres") +
  coord_flip(clip = "off") +
  theme_bw() +
  scale_y_continuous(labels = comma,
                     limits = c(-1000, 2500000))
